CONVOLUTIONS ARE ABOUT THE RELATIONSHIPS BETWEEN A PIXEL AND ITS NEIGHBORS

MultiPly and add a pixel and its neighbors by a set of weights

applying A convolution

import numpy as np
import matplotlib.pyplot as plt

im = plt.imread('../images/grid.jpg') # grayscale!
plt.imshow(im,cmap='gray')

from scipy.signal import convolve2d

# vertical edge detector
vert_kernel = np.array([[1, 0, -1], 
                        [1, 0, -1],
                        [1, 0, -1]])

vertical = convolve2d(im, vert_kernel, 'same')
plt.imshow(vertical,cmap='gray')

convolutionS WORK WITH ROTATIONS

# horizontal edge detector
# by rotating our vertical filter 90 degrees
# [-1 -1 -1]
# [0  0  0]
# [1  1  1]

horiz_kernel = np.rot90(vert_kernel)
horizontal = convolve2d(im, horiz_kernel, 'same')

plt.imshow(horizontal,cmap='gray')

convolutionS CAN DETECT EDGES

# Example edge detector
# [-1 -1 -1]
# [-1 8  -1]
# [-1 -1 -1]

edge_kernel = np.array([[-1, -1, -1],
                        [-1, 8, -1],
                        [-1, -1, -1]])

edge = convolve2d(im, edge_kernel, 'same')
plt.figure()
plt.imshow(edge,cmap='gray')

convolutionS CAN BLUR

# Example blur kernel
# [.0625 .125 .0625]
# [.125 .25  .125]
# [.0625 .125 .0625]

blur_kernel = np.array([[0.0625, 0.125, 0.0625],
                        [0.125, 0.25, 0.125],
                        [0.0625, 0.125, 0.0625]])

blur = convolve2d(im, blur_kernel, 'same')
plt.figure()
plt.imshow(blur,cmap='gray')
